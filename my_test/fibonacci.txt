    var numero ^= 1;\
    <-- "Inserisci numero intero: " $(numero);
    --> $(fibonacci(numero));
endproc

func fibonacci(numero : integer) -> integer:

        var i ^= 1; previousPreviousNumber, previousNumber ^= 0, 0; currentNumber ^= 1;\

        while i < numero do
            previousPreviousNumber ^= previousNumber;
            previousNumber ^= currentNumber;
            currentNumber ^= previousPreviousNumber + previousNumber;
            i ^= i + 1;
        endwhile;

        return currentNumber;

endfunc
