%questo programma mostra lo scoping del linguaggio%

var stringa ^= "globale";\

proc main():
    var stringa ^= "locale";\
    -->! "localscope: " $(stringa);
    globalstampa();
endproc

proc globalstampa():

    -->! "globalscope: " $(stringa);

    if true then
        var stringa ^= "stringa if";\
        -->! "ifscope: " $(stringa);
    endif;

    var contatore ^=2;\
    while contatore > 0 do
        var stringa ^= "stringa while";\
        -->! "whilescope: " $(stringa);
        contatore ^= 0;
    endwhile;

endproc
